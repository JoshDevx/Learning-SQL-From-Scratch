# Filtrar Texto

## .Buscar patrones
LIKE
% - Coincide con cero, uno o muchos caracteres
SELECT campo
FROM tabla
WHERE campo LIKE 'Ade%'; - '%r'(Terminen con r)
(_) Coincide con un solo caracter
SELECT campo
FROM tabla
WHERE campo LIKE 'Ade_'; - '__t%' (Tenga la T en posicion 3 y despues cualquier caracter)

NOT LIKE
Encontrar registros que no coincidan
SELECT campo
FROM tabla
WHERE campo NOT LIKE 'A.%';
'No coinciden con A.'

IN
SELECT campo
FROM tabla
WHERE campo IN (1920, 1930, 1940);

En lugar de usar OR varias veces, podemos usar IN

SELECT campo
FROM tabla
WHERE campo IN ('Texto', 'Texto', 'Texto');
